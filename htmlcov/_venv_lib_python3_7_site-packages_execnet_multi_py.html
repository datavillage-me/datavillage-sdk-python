98 def makegateway(self, spec=None): 

99 """create and configure a gateway to a Python interpreter. 

100 The ``spec`` string encodes the target gateway type 

101 and configuration information. The general format is:: 

102 

103 key1=value1//key2=value2//... 

104 

105 If you leave out the ``=value`` part a True value is assumed. 

106 Valid types: ``popen``, ``ssh=hostname``, ``socket=host:port``. 

107 Valid configuration:: 

108 

109 id=<string> specifies the gateway id 

110 python=<path> specifies which python interpreter to execute 

111 execmodel=model 'thread', 'eventlet', 'gevent' model for execution 

112 chdir=<path> specifies to which directory to change 

113 nice=<path> specifies process priority of new process 

114 env:NAME=value specifies a remote environment variable setting. 

115 

116 If no spec is given, self.defaultspec is used. 

117 """ 

118 if not spec: 

119 spec = self.defaultspec 

120 if not isinstance(spec, XSpec): 

121 spec = XSpec(spec) 

122 self.allocate_id(spec) 

123 if spec.execmodel is None: 

124 spec.execmodel = self.remote_execmodel.backend 

125 if spec.via: 

126 assert not spec.socket 

127 master = self[spec.via] 

128 proxy_channel = master.remote_exec(gateway_io) 

129 proxy_channel.send(vars(spec)) 

130 proxy_io_master = gateway_io.ProxyIO(proxy_channel, self.execmodel) 

131 gw = gateway_bootstrap.bootstrap(proxy_io_master, spec) 

132 elif spec.popen or spec.ssh or spec.vagrant_ssh: 

133 io = gateway_io.create_io(spec, execmodel=self.execmodel) 

134 gw = gateway_bootstrap.bootstrap(io, spec) 

135 elif spec.socket: 

136 from execnet import gateway_socket 

137 

138 io = gateway_socket.create_io(spec, self, execmodel=self.execmodel) 

139 gw = gateway_bootstrap.bootstrap(io, spec) 

140 else: 

141 raise ValueError("no gateway type found for {!r}".format(spec._spec)) 

142 gw.spec = spec 

143 self._register(gw) 

144 if spec.chdir or spec.nice or spec.env: 

145 channel = gw.remote_exec( 

146 """ 

147 import os 

148 path, nice, env = channel.receive() 

149 if path: 

150 if not os.path.exists(path): 

151 os.mkdir(path) 

152 os.chdir(path) 

153 if nice and hasattr(os, 'nice'): 

154 os.nice(nice) 

155 if env: 

156 for name, value in env.items(): 

157 os.environ[name] = value 

158 """ 

159 ) 

160 nice = spec.nice and int(spec.nice) or 0 

161 channel.send((spec.chdir, nice, spec.env)) 

162 channel.waitclose() 

163 return gw

165 def allocate_id(self, spec): 

166 """ (re-entrant) allocate id for the given xspec object. """ 

167 if spec.id is None: 

168 with self._autoidlock: 

169 id = "gw" + str(self._autoidcounter)

172 raise ValueError("already have gateway with id {!r}".format(id))

190 def terminate(self, timeout=None): 

191 """ trigger exit of member gateways and wait for termination 

192 of member gateways and associated subprocesses. After waiting 

193 timeout seconds try to to kill local sub processes of popen- 

194 and ssh-gateways. Timeout defaults to None meaning 

195 open-ended waiting and no kill attempts. 

196 """ 

197 

198 while self: 

199 vias = {} 

200 for gw in self: 

201 if gw.spec.via: 

202 vias[gw.spec.via] = True 

203 for gw in self: 

204 if gw.id not in vias: 

205 gw.exit()

299 def termkill(termfunc, killfunc): 

300 termreply = workerpool.spawn(termfunc) 

301 try: 

302 termreply.get(timeout=timeout) 

303 except IOError:

306 replylist = [] 

307 for termfunc, killfunc in list_of_paired_functions: 

308 reply = workerpool.spawn(termkill, termfunc, killfunc)
